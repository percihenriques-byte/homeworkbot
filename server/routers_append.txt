
  integrationSettings: router({
    get: protectedProcedure.query(async ({ ctx }) => {
      return await db.getIntegrationSettings(ctx.user.id);
    }),
    update: protectedProcedure
      .input(z.object({
        emailSmtpHost: z.string().optional(),
        emailSmtpPort: z.number().optional(),
        emailUsername: z.string().optional(),
        emailPassword: z.string().optional(),
        emailSenderName: z.string().optional(),
        emailSenderEmail: z.string().optional(),
        toddleApiKey: z.string().optional(),
        toddleEnabled: z.boolean().optional(),
        whatsappPhoneNumber: z.string().optional(),
        whatsappApiKey: z.string().optional(),
        whatsappEnabled: z.boolean().optional(),
      }))
      .mutation(async ({ ctx, input }) => {
        return await db.createOrUpdateIntegrationSettings(ctx.user.id, input);
      }),
  }),
});
